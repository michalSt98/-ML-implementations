class GA(object):
    
    def __init__(self, pop_size, rectangles, r):
        self.pop_size = pop_size #Population size
        self.rectangles = rectangles #Different rectangles
        self.r = r #Radius of circle in (0,0)
        
        self.population = [] #Random initialisation
        
        for i in range(pop_size):
            rec = self.rectangle()
            while(not self.check_if_in_circle(rec)):
                rec = self.rectangle()
            self.population.append([rec])
            
        
        self.scores = []
        self.average_scores = []
        self.best_scores = []
        self.when_best = []
        self.best_genotype = []
        self.best_scores.append(np.Inf)
        self.epoch = 0
        
    
    def rectangle(self):
        bottomleftx = np.random.uniform(low = -self.r *1.0, high = self.r*1.0)
        bottomlefty = np.random.uniform(low = -self.r *1.0, high = self.r*1.0)
        l = self.rectangles.shape[0]
        rtype = np.random.choice(l)
        return [bottomleftx, bottomlefty, rtype]
    
    def check_if_in_circle(self, rec):
        x = rec[0]
        y = rec[1]
        width = self.rectangles[rec[2]][0]
        height = self.rectangles[rec[2]][1]
        
        for a in (x, x+width):
            for b in (y, y+height):
                # compare distance between circle's center point and each point of
                # the rectangle with the circle's radius
                if math.hypot(a, b) >= self.r:
                    return False  # collision detected
        return True
    
    def check_collision_with_rect(self, genotype, rec):
        rec_x = rec[0]
        rec_y = rec[1]
        rec_width = self.rectangles[rec[2]][0]
        rec_height = self.rectangles[rec[2]][1]
        for subject in genotype:
            sub_x = subject[0]
            sub_y = subject[1]
            sub_width = self.rectangles[subject[2]][0]
            sub_height = self.rectangles[subject[2]][1]
            if not (sub_x + sub_width  < rec_x or sub_x > rec_x + rec_width or sub_y + sub_height < rec_y or sub_y > rec_y + rec_height):
                return True
        return False  
        
    def pop_scores(self): #Function that updates scores
        scores = []
        for genotype in self.population:
            value = 0
            for rec in genotype:
                value += self.rectangles[rec[2]][2]
            scores.append(-value)
        self.scores.append(scores)
        self.average_scores.append(np.mean(self.scores[-1]))
        
        candidate = np.min(self.scores[-1])
        if(candidate < self.best_scores[-1]):
            self.best_scores.append(candidate)
            self.when_best.append(self.epoch)
            self.best_genotype.append(self.population[np.argmin(self.scores[-1])])
        
    def mutation(self): #Our mutation is adding or removing random rectangle from genotype
        for genotype_n in range(len(self.offspring)):
            seed = np.random.uniform(0,1)
            if(seed < self.mut_prob):
                for i in range(50):
                    rec = self.rectangle()
                    col = self.check_if_in_circle(rec)
                    if((not self.check_collision_with_rect(self.offspring[genotype_n], rec)) and col):
                        self.offspring[genotype_n].append(rec)
                        break
        for genotype_n in range(len(self.population)):
            seed = np.random.uniform(0,1)
            if(seed < self.mut_prob):
                for i in range(50):
                    rec = self.rectangle()
                    col = self.check_if_in_circle(rec)
                    if((not self.check_collision_with_rect(self.population[genotype_n], rec)) and col):
                        self.population[genotype_n].append(rec)
                        break
                        
        for genotype_n in range(len(self.population)):
            seed = np.random.uniform(0,1)
            if(seed/2 < self.mut_prob):
                if(len(self.population[genotype_n])>0):
                    idx = np.random.choice(len(self.population[genotype_n]))
                    self.population[genotype_n].remove(self.population[genotype_n][idx])
                
    def crossover(self): # Our crossover is summing parents rectangles and adding them to offspring
        #at random, it can be better optimised
        self.pop_scores()
        p = (1e-6 +np.max(self.scores[-1])-self.scores[-1])/(1e-6 + np.sum(np.max(self.scores[-1]) - self.scores[-1]))
        p = p/(sum(p)) #Pobabilities of best genotypes
        n = len(self.population)
        k = int(n*self.cross_prob)
        if(k%2 != 0):
            k += 1
        parents_indx = np.random.choice(n, k, p = p, replace = False)

        self.offspring = []
        i = -1
        while(i+2<len(parents_indx)):
            i += 1
            a = self.population[parents_indx[i]]
            b = self.population[parents_indx[i+1]]
            every = a + b
            np.random.shuffle(every)
            new = []
            new.append(every[0])
            for j in range(len(every)-1):
                j = j+1
                if(not self.check_collision_with_rect(new, every[j])):
                    new.append(every[j])
            self.offspring.append(new)
            
            i += 1
            
    def fit(self, n_epochs, mut_prob = 0.2, cross_prob = 0.7):
        self.n_epochs = n_epochs
        self.mut_prob = mut_prob
        self.epoch = 0
        self.cross_prob = cross_prob
        
        self.pop_scores()
        for epoch in range(self.n_epochs):
            self.epoch = epoch*2
            if(self.best_scores == 0):
                break
            self.crossover()
            self.mutation()
            self.population = self.population + self.offspring
            self.pop_scores()
            
            p = (1e-6 +np.max(self.scores[-1])-self.scores[-1])/(1e-6 +np.sum(np.max(self.scores[-1]) - self.scores[-1]))
            p = p/(sum(p))
            n = len(self.population)
            k = self.pop_size
            winners_indx = np.random.choice(n, k, p = p, replace = False)
            self.population = [self.population[i] for i in winners_indx]
